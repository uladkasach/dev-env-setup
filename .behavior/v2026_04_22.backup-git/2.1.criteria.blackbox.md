# criteria.blackbox: git backup

## episode experience

# usecase.1 = backup all repos and claude sessions

given('user has ~/git/ directory with repos')
given('user has ~/.claude/ directory')
given('user has aws credentials configured')
  when('user runs `git backup --repo all --into s3://bucket/machine=name`')
    then('displays repo tree status showing all repos and worktrees')
      sothat('user sees what will be backed up')
    then('displays branch status for each worktree (unstaged, merged, etc)')
      sothat('user knows the state of their work')
    then('creates git.tar.gz from ~/git/ excluding node_modules/ and .cache/')
    then('creates .claude.tar.gz from ~/.claude/')
    then('uploads both archives to s3://bucket/machine=name/effectiveAt=$timestamp/~/')
    then('displays confirmation with 🍯 got em header')
    then('displays exact S3 URIs and file sizes for each archive')
      sothat('user can verify backup location and size')

## exchange experience

# usecase.2 = missing git directory

given('user has no ~/git/ directory')
  when('user runs `git backup --repo all --into s3://...`')
    then('displays "no git dir found" message')
    then('exits with non-zero status')
      sothat('user knows no repos exist to backup')

# usecase.3 = missing claude directory

given('user has ~/git/ directory')
given('user has no ~/.claude/ directory')
  when('user runs `git backup --repo all --into s3://...`')
    then('creates and uploads git.tar.gz')
    then('displays warning that ~/.claude/ not found')
    then('skips .claude.tar.gz creation')
    then('displays confirmation for git.tar.gz only')
      sothat('user knows claude sessions were not backed up')

# usecase.4 = missing required arguments

given('user runs git backup without --repo flag')
  when('command is executed')
    then('displays error indicating --repo is required')
    then('exits with non-zero status')

given('user runs git backup without --into flag')
  when('command is executed')
    then('displays error indicating --into is required')
    then('exits with non-zero status')

# usecase.5 = invalid s3 uri

given('user provides --into without s3:// prefix')
  when('user runs `git backup --repo all --into bucket/path`')
    then('displays error indicating invalid S3 URI format')
    then('exits with non-zero status')

# usecase.6 = s3 upload failure

given('user has valid local directories')
given('aws credentials are expired or invalid')
  when('user runs `git backup --repo all --into s3://...`')
    then('displays repo tree status')
    then('creates tar.gz archives locally')
    then('displays error from aws cli on upload failure')
    then('exits with non-zero status')
      sothat('user knows backup did not complete')

# usecase.7 = output path structure

given('user provides --into s3://vlad-backups/machine=vlad.laptop1')
  when('backup completes')
    then('git.tar.gz uploaded to s3://vlad-backups/machine=vlad.laptop1/effectiveAt=$timestamp/~/git.tar.gz')
    then('.claude.tar.gz uploaded to s3://vlad-backups/machine=vlad.laptop1/effectiveAt=$timestamp/~/.claude.tar.gz')
    then('$timestamp is ISO8601 format like 2026-04-22T14:30:00Z')
      sothat('backups are organized by time and restorable to ~/')
